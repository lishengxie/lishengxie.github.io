cd /etc/apt
# 将原来的源文件复制一份做备份
sudo cp sources.list sources.list.cpk
sudo vim sources.list
# 删除全部内容换成上面的镜像网址(具体操作为Esc->:%d->Enter->i进入插入模式->粘贴内容->Esc->:wq)
# 保存退出后更新
sudo apt-get update
sudo apt-get upgrade
